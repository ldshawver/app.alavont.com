🧱 STEP 1 — Core Print Function (Failover Logic)

This is the brain. Put this in your backend (Replit).

printService.js
const fetch = require("node-fetch");

const DIRECT_PRINTER_IP = "192.168.68.66";
const BRIDGE_URL = "http://192.168.68.82:3001";
const API_KEY = "change_this_secret";

// Simple network print via IPP (fallback to system command if needed)
async function printDirect(text) {
  try {
    const res = await fetch(`http://${DIRECT_PRINTER_IP}:631/ipp/print`, {
      method: "POST",
      body: text,
      timeout: 3000
    });

    if (!res.ok) throw new Error("Direct print failed");

    return true;
  } catch (err) {
    throw new Error("Direct printer unreachable");
  }
}

// Mac bridge print
async function printViaBridge(text) {
  const res = await fetch(`${BRIDGE_URL}/print`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "x-api-key": API_KEY
    },
    body: JSON.stringify({ text }),
    timeout: 5000
  });

  const data = await res.json();

  if (!data.success) {
    throw new Error(data.error || "Bridge failed");
  }

  return true;
}

// MAIN FAILOVER FUNCTION
async function printOrder(orderText) {
  try {
    console.log("Trying direct printer...");
    await printDirect(orderText);
    return { success: true, method: "direct" };

  } catch (err1) {
    console.log("Direct failed, trying bridge...");

    try {
      await printViaBridge(orderText);
      return { success: true, method: "bridge" };

    } catch (err2) {
      console.log("Bridge failed, queueing...");
      return { success: false, method: "queued" };
    }
  }
}

module.exports = { printOrder };
🧾 STEP 2 — Add Print Queue (CRITICAL)

If both methods fail, we store the job.

Simple version (start here)

Create file:

printQueue.js
const fs = require("fs");
const path = require("path");

const QUEUE_FILE = path.join(__dirname, "print-queue.json");

function loadQueue() {
  if (!fs.existsSync(QUEUE_FILE)) return [];
  return JSON.parse(fs.readFileSync(QUEUE_FILE));
}

function saveQueue(queue) {
  fs.writeFileSync(QUEUE_FILE, JSON.stringify(queue, null, 2));
}

function addToQueue(text) {
  const queue = loadQueue();
  queue.push({
    id: Date.now(),
    text,
    retries: 0,
    createdAt: new Date().toISOString()
  });
  saveQueue(queue);
}

module.exports = { loadQueue, saveQueue, addToQueue };
🔁 STEP 3 — Retry Worker (AUTO PRINT LATER)

This runs in background and retries failed jobs.

printWorker.js
const { loadQueue, saveQueue } = require("./printQueue");
const { printOrder } = require("./printService");

async function processQueue() {
  let queue = loadQueue();

  if (queue.length === 0) return;

  console.log(`Processing ${queue.length} queued jobs...`);

  const remaining = [];

  for (const job of queue) {
    try {
      const result = await printOrder(job.text);

      if (!result.success) {
        job.retries++;
        if (job.retries < 5) {
          remaining.push(job);
        }
      }

    } catch (err) {
      job.retries++;
      if (job.retries < 5) {
        remaining.push(job);
      }
    }
  }

  saveQueue(remaining);
}

// run every 10 seconds
setInterval(processQueue, 10000);

console.log("Print worker running...");
