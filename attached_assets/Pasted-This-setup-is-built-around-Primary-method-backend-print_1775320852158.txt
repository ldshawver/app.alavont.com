services/receiptRenderer.js
function money(n) {
  return `$${Number(n || 0).toFixed(2)}`;
}

function padRight(str, len) {
  str = String(str || "");
  return str.length >= len ? str.slice(0, len) : str + " ".repeat(len - str.length);
}

function renderKitchenTicket(order) {
  const lines = [];
  lines.push("================================");
  lines.push("           NEW ORDER            ");
  lines.push("================================");
  lines.push(`Order #: ${order.orderNumber}`);
  lines.push(`Type: ${order.fulfillmentType || "Pickup"}`);
  lines.push(`Time: ${new Date(order.createdAt).toLocaleString()}`);
  if (order.customerName) lines.push(`Name: ${order.customerName}`);
  if (order.notes) {
    lines.push("--------------------------------");
    lines.push("ORDER NOTES:");
    lines.push(order.notes);
  }
  lines.push("--------------------------------");

  for (const item of order.items || []) {
    lines.push(`${item.quantity} x ${item.name}`);
    for (const mod of item.modifiers || []) {
      lines.push(`  + ${mod.name}`);
    }
    if (item.notes) {
      lines.push(`  * ${item.notes}`);
    }
  }

  lines.push("--------------------------------");
  lines.push(`Payment: ${order.paymentStatus || "Pending"}`);
  lines.push("================================");
  lines.push("");
  lines.push("");
  return lines.join("\n");
}

function renderCustomerReceipt(order) {
  const lines = [];
  lines.push("================================");
  lines.push("          ORDER RECEIPT         ");
  lines.push("================================");
  lines.push(`Order #: ${order.orderNumber}`);
  lines.push(`Date: ${new Date(order.createdAt).toLocaleString()}`);
  lines.push("--------------------------------");

  for (const item of order.items || []) {
    const qtyName = `${item.quantity} x ${item.name}`;
    const price = money((item.price || 0) * (item.quantity || 1));
    lines.push(`${padRight(qtyName, 26)}${price}`);
    for (const mod of item.modifiers || []) {
      lines.push(`  + ${mod.name}`);
    }
  }

  lines.push("--------------------------------");
  lines.push(`${padRight("Subtotal", 26)}${money(order.subtotal)}`);
  lines.push(`${padRight("Tax", 26)}${money(order.tax)}`);
  lines.push(`${padRight("Total", 26)}${money(order.total)}`);
  lines.push("--------------------------------");
  lines.push(`Payment: ${order.paymentStatus || "Pending"}`);
  lines.push("");
  lines.push("Thank you!");
  lines.push("");
  lines.push("");
  return lines.join("\n");
}

module.exports = {
  renderKitchenTicket,
  renderCustomerReceipt,
};
services/printJobService.js
const crypto = require("crypto");

function makeIdempotencyKey(orderId, printerId, jobType) {
  return crypto
    .createHash("sha256")
    .update(`${orderId}:${printerId}:${jobType}`)
    .digest("hex");
}

async function createPrintJob(db, { orderId, printerId, jobType, payloadJson }) {
  const idempotencyKey = makeIdempotencyKey(orderId, printerId, jobType);

  const existing = await db.query(
    `SELECT id FROM print_jobs WHERE idempotency_key = $1 LIMIT 1`,
    [idempotencyKey]
  );

  if (existing.rows.length) {
    return existing.rows[0];
  }

  const result = await db.query(
    `INSERT INTO print_jobs
      (order_id, printer_id, job_type, status, idempotency_key, payload_json)
     VALUES ($1, $2, $3, 'queued', $4, $5)
     RETURNING *`,
    [orderId, printerId, jobType, idempotencyKey, payloadJson]
  );

  return result.rows[0];
}

module.exports = { createPrintJob };

const fetch = require("node-fetch");

async function sleep(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

function getBackoffMs(retryCount, base = 3000) {
  return Math.min(base * Math.pow(2, retryCount), 60000);
}

async function dispatchOne(db, job, printer) {
  await db.query(
    `UPDATE print_jobs SET status = 'sending', last_attempt_at = NOW(), updated_at = NOW()
     WHERE id = $1`,
    [job.id]
  );

  try {
    const res = await fetch(`${printer.bridge_url}/print`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "x-api-key": process.env.PRINT_BRIDGE_API_KEY,
      },
      body: JSON.stringify({
        printerName: printer.bridge_printer_name || printer.name,
        jobId: job.id,
        format: job.render_format,
        text: job.rendered_text,
        payload: job.payload_json,
        copies: printer.copies || 1,
      }),
      timeout: printer.timeout_ms || 8000,
    });

    const data = await res.json();

    await db.query(
      `INSERT INTO print_job_attempts
        (print_job_id, attempt_number, request_payload, response_payload, success, error_message)
       VALUES ($1, $2, $3, $4, $5, $6)`,
      [
        job.id,
        job.retry_count + 1,
        { printer: printer.name },
        data,
        !!data.success,
        data.error || null,
      ]
    );

    if (!data.success) {
      throw new Error(data.error || "Unknown print error");
    }

    await db.query(
      `UPDATE print_jobs
       SET status = 'printed', printed_at = NOW(), updated_at = NOW()
       WHERE id = $1`,
      [job.id]
    );
  } catch (err) {
    const newRetryCount = job.retry_count + 1;
    const maxRetries = job.max_retries || 5;

    await db.query(
      `UPDATE print_jobs
       SET status = $2, retry_count = $3, error_message = $4, updated_at = NOW()
       WHERE id = $1`,
      [
        job.id,
        newRetryCount >= maxRetries ? "failed" : "retrying",
        newRetryCount,
        err.message,
      ]
    );
  }
}

async function startPrintWorker(db) {
  while (true) {
    try {
      const jobs = await db.query(`
        SELECT j.*, p.*
        FROM print_jobs j
        JOIN print_printers p ON p.id = j.printer_id
        WHERE j.status IN ('queued', 'retrying')
          AND p.is_active = true
        ORDER BY j.created_at ASC
        LIMIT 10
      `);

      for (const row of jobs.rows) {
        const job = {
          id: row.id,
          retry_count: row.retry_count,
          max_retries: row.max_retries,
          render_format: row.render_format,
          rendered_text: row.rendered_text,
          payload_json: row.payload_json,
        };

        const printer = {
          name: row.name,
          bridge_url: row.bridge_url,
          bridge_printer_name: row.bridge_printer_name,
          timeout_ms: row.timeout_ms,
          copies: row.copies,
        };

        await dispatchOne(db, job, printer);
      }
    } catch (err) {
      console.error("Print worker error:", err);
    }

    await sleep(3000);
  }
}

module.exports = { startPrintWorker };

const { renderKitchenTicket, renderCustomerReceipt } = require("./services/receiptRenderer");
const { createPrintJob } = require("./services/printJobService");

async function onOrderCreated(db, order) {
  const printers = await db.query(
    `SELECT * FROM print_printers WHERE is_active = true`
  );

  for (const printer of printers.rows) {
    let renderedText = null;
    let jobType = "order_ticket";

    if (printer.role === "kitchen") {
      renderedText = renderKitchenTicket(order);
    } else if (printer.role === "receipt") {
      renderedText = renderCustomerReceipt(order);
      jobType = "customer_receipt";
    } else {
      continue;
    }

    const job = await createPrintJob(db, {
      orderId: order.id,
      printerId: printer.id,
      jobType,
      payloadJson: order,
    });

    await db.query(
      `UPDATE print_jobs SET rendered_text = $2, updated_at = NOW() WHERE id = $1`,
      [job.id, renderedText]
    );
  }
}

const express = require("express");
const { exec } = require("child_process");
const fs = require("fs");
const os = require("os");
const path = require("path");

let escpos = null;
try {
  escpos = require("escpos");
  escpos.USB = require("escpos-usb");
} catch (e) {
  console.warn("ESC/POS modules unavailable, CUPS fallback only.");
}

const app = express();
app.use(express.json({ limit: "1mb" }));

const API_KEY = process.env.PRINT_BRIDGE_API_KEY || "change-me";

function auth(req, res, next) {
  if (req.headers["x-api-key"] !== API_KEY) {
    return res.status(403).json({ success: false, error: "Forbidden" });
  }
  next();
}

function printViaEscpos(text, copies = 1) {
  return new Promise((resolve, reject) => {
    try {
      const device = new escpos.USB();
      const printer = new escpos.Printer(device);

      device.open(err => {
        if (err) return reject(err);

        let chain = printer.align("lt");
        for (let i = 0; i < copies; i++) {
          chain = chain.text(text).text("\n\n").cut();
        }
        chain.close(() => resolve({ method: "escpos" }));
      });
    } catch (err) {
      reject(err);
    }
  });
}

function printViaCups(text, printerName, copies = 1) {
  return new Promise((resolve, reject) => {
    const filePath = path.join(os.tmpdir(), `print-${Date.now()}.txt`);
    fs.writeFileSync(filePath, text, "utf8");

    const cmd = printerName
      ? `lp -d "${printerName}" -n ${copies} "${filePath}"`
      : `lp -n ${copies} "${filePath}"`;

    exec(cmd, (err, stdout, stderr) => {
      fs.unlink(filePath, () => {});
      if (err) return reject(new Error(stderr || err.message));
      resolve({ method: "cups", stdout: stdout.trim() });
    });
  });
}

app.get("/health", (req, res) => {
  exec("lpstat -p -d", (err, stdout, stderr) => {
    res.json({
      success: !err,
      printerStatus: stdout || stderr,
      hostname: os.hostname(),
      time: new Date().toISOString(),
    });
  });
});

app.post("/print", auth, async (req, res) => {
  const { printerName, text, copies } = req.body;

  if (!text || typeof text !== "string") {
    return res.status(400).json({ success: false, error: "Missing text payload" });
  }

  try {
    if (escpos) {
      const result = await printViaEscpos(text, copies || 1);
      return res.json({ success: true, result });
    }
  } catch (err) {
    console.warn("ESC/POS failed, trying CUPS:", err.message);
  }

  try {
    const result = await printViaCups(text, printerName, copies || 1);
    return res.json({ success: true, result });
  } catch (err) {
    return res.status(500).json({ success: false, error: err.message });
  }
});

const port = process.env.PORT || 3001;
app.listen(port, () => {
  console.log(`Print bridge running on ${port}`);
});
